Conditional Statements

// If-else
if (age >= 18) {
    echo "Adult";
} else {
    echo "Minor";
}

// Switch
switch(day) {
    case "Monday":
        echo "Start of week";
        break;
    default:
        echo "Other day";
}

Loops in PHP

// For loop
for ($i = 0; $i < 5; $i++) {
    echo $i;
}

// While loop
$i = 0;
while ($i < 5) {
    echo $i;
    $i++;
}

// Foreach loop
$colors = ["red", "green", "blue"];
foreach ($colors as $color) {
    echo $color;
}

Arrays

// Indexed array
$fruits = ["Apple", "Banana", "Orange"];

// Associative array
$person = [
    "name" => "John",
    "age" => 25,
    "city" => "New York"
];
